namespace DesktopCompanion
{
    public partial class MainWindow : Window
    {
        private readonly HttpClient _httpClient = new HttpClient();
        private string _pythonBackendUrl = "http://localhost:8000";

        public MainWindow()
        {
            InitializeComponent();
            InitializeWebView();
        }

        private async void InitializeWebView()
        {
            await WebView.EnsureCoreWebView2Async(null);
            WebView.CoreWebView2.WebMessageReceived += HandleWebMessage;
            WebView.CoreWebView2.Navigate("http://localhost:3000"); // React dev server
        }

        private async void HandleWebMessage(object sender, CoreWebView2WebMessageReceivedEventArgs e)
        {
            var messageJson = e.TryGetWebMessageAsString();
            var message = JsonDocument.Parse(messageJson).RootElement;

            if (message.GetProperty("action").GetString() == "capture_and_ask")
            {
                string prompt = message.GetProperty("prompt").GetString();
                string base64Image = CaptureScreen();

                // Send to Python Backend and stream response back to React
                await StreamFromBackendAsync(prompt, base64Image);
            }
        }

        private string CaptureScreen()
        {
            // Simplified capture using System.Drawing
            var bounds = System.Windows.Forms.Screen.PrimaryScreen.Bounds;
            using var bitmap = new Bitmap(bounds.Width, bounds.Height);
            using var g = Graphics.FromImage(bitmap);
            g.CopyFromScreen(bounds.Location, Point.Empty, bounds.Size);

            using var ms = new MemoryStream();
            bitmap.Save(ms, ImageFormat.Jpeg); // JPEG for smaller payload
            return Convert.ToBase64String(ms.ToArray());
        }

        private async Task StreamFromBackendAsync(string prompt, string base64Image)
        {
            var payload = new
            {
                prompt = prompt,
                image_base64 = base64Image,
                history = new object[] { } // In-session history managed by React, passed here if needed
            };

            var content = new StringContent(JsonSerializer.Serialize(payload), Encoding.UTF8, "application/json");
            var response = await _httpClient.PostAsync($"{_pythonBackendUrl}/chat", content, HttpCompletionOption.ResponseHeadersRead);

            using var stream = await response.Content.ReadAsStreamAsync();
            using var reader = new StreamReader(stream);

            while (!reader.EndOfStream)
            {
                var line = await reader.ReadLineAsync();
                if (line != null)
                {
                    // Forward streamed chunk to React UI
                    string script = $"window.dispatchEvent(new CustomEvent('ai-chunk', {{ detail: '{line.Replace("'", "\\'")}' }}));";
                    WebView.CoreWebView2.ExecuteScriptAsync(script);
                }
            }
        }
    }

**main.py (FastAPI Backend)**
```python
from fastapi import FastAPI, Request
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
from openai import OpenAI
import json
import uvicorn

app = FastAPI()

# Configure for local LLM (Ollama or LM Studio)
client = OpenAI(
    base_url="http://localhost:11434/v1", # Ollama default
    api_key="not-needed"
)

class ChatRequest(BaseModel):
    prompt: str
    image_base64: str
    history: list = []

@app.post("/chat")
async def chat(req: ChatRequest):
    def event_stream():
        # Construct multimodal payload
        messages = [
            {
                "role": "user",
                "content": [
                    {"type": "text", "text": req.prompt},
                    {
                        "type": "image_url",
                        "image_url": f"data:image/jpeg;base64,{req.image_base64}"
                    }
                ]
            }
        ]
        
        # Add history if provided (for follow-up questions in the same session)
        messages = req.history + messages

        try:
            # Call vision-capable LLM (e.g., LLaVA via Ollama)
            response = client.chat.completions.create(
                model="llava",
                messages=messages,
                stream=True
            )
            
            for chunk in response:
                if chunk.choices[0].delta.content is not None:
                    # Format as Server-Sent Events (SSE) for easy parsing in C#
                    data = json.dumps({"text": chunk.choices[0].delta.content})
                    yield f"data: {data}\n\n"
                    
        except Exception as e:
            yield f"data: {json.dumps({'error': str(e)})}\n\n"

    return StreamingResponse(event_stream(), media_type="text/event-stream")

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

// Services/AIWebSocketClient.cs
using System.Net.WebSockets;

public class AIWebSocketClient
{
    private ClientWebSocket _ws = new ClientWebSocket();

    public async Task ConnectAsync(string uri = "ws://localhost:8000/ws")
    {
        await _ws.ConnectAsync(new Uri(uri), CancellationToken.None);
        _ = ListenAsync(); // Start background listener for AI responses
    }

    // Called by WebViewBridge when React sends a message
    public async Task SendVisionRequestAsync(string prompt, byte[] imageBytes)
    {
        // In 2026, we send binary frames, not base64 JSON, to save 33% bandwidth
        var payload = new { type = "vision_prompt", prompt = prompt };
        var jsonPayload = JsonSerializer.SerializeToUtf8Bytes(payload);
        
        // Send JSON metadata frame
        await _ws.SendAsync(jsonPayload, WebSocketMessageType.Text, true, CancellationToken.None);
        // Send Binary Image frame directly
        await _ws.SendAsync(imageBytes, WebSocketMessageType.Binary, true, CancellationToken.None);
    }

    private async Task ListenAsync()
    {
        var buffer = new byte[8192];
        while (_ws.State == WebSocketState.Open)
        {
            var result = await _ws.ReceiveAsync(new ArraySegment<byte>(buffer), CancellationToken.None);
            var chunk = System.Text.Encoding.UTF8.GetString(buffer, 0, result.Count);
            
            // Stream directly to React via strongly typed bridge
            WebViewBridge.DispatchAIChunk(chunk);
        }
    }
}
